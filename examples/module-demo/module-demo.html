2. 代码实现

父页面代码：

// 从 SDK 构建产物中导入 PostMessageCommAPI
import { PostMessageCommAPI } from '../encrypted-chat-sdk/dist/index.js';

// 初始化通信 API
const iframe = document.getElementById('demoFrame');
const commAPI = new PostMessageCommAPI(iframe.contentWindow, '*');

// 连接通信
commAPI.connect().then(() => {
    logMessage('系统', '已连接到 iframe');
});

// 监听消息
commAPI.on('newMessage', (message) => {
    logMessage(message.from, message.content, 'received');
});

// 发送消息
document.getElementById('sendButton').addEventListener('click', () => {
    const content = document.getElementById('messageInput').value;
    if (content.trim()) {
        commAPI.send(content).then(() => {
            logMessage('我', content, 'sent');
            document.getElementById('messageInput').value = '';
        });
    }
});

3. 使用说明

  1. 确保已构建 SDK：cd encrypted-chat-sdk && npm run build
  2. 使用 ES 模块导入方式：import { PostMessageCommAPI } from './path/to/sdk/dist/index.js';
  3. 初始化 API：const commAPI = new PostMessageCommAPI(target, origin);
  4. 连接：await commAPI.connect();
  5. 监听消息：commAPI.on('newMessage', callback);
  6. 发送消息：await commAPI.send('消息内容');

注意： 在生产环境中，请将 '*' 替换为具体的 origin 并进行来源校验。
